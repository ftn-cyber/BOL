<script>
let paketDipilih = "";
let hargaPaket = 0;
let detailPaket = "";
let fiturTambahan = [];
let totalTambahan = 0;
let rankDipilih = "";
let mapDipilih = "";

function pilihPaket(nama,harga,detail){

paketDipilih = nama;
hargaPaket = harga;
detailPaket = detail;

document.querySelectorAll('.card').forEach(card=>{
card.classList.remove('active');
});

event.currentTarget.classList.add('active');
}

function toggleFeature(el,nama,harga){

if(el.classList.contains('active')){
el.classList.remove('active');

fiturTambahan = fiturTambahan.filter(f=>f.nama !== nama);
}else{
el.classList.add('active');

fiturTambahan.push({
nama:nama,
harga:harga
});
}

hitungTambahan();
}

function hitungTambahan(){
totalTambahan = 0;

fiturTambahan.forEach(f=>{
totalTambahan += f.harga;
});
}

document.querySelectorAll('#rankOptions .option').forEach(opt=>{
opt.addEventListener('click',function(){

document.querySelectorAll('#rankOptions .option').forEach(o=>{
o.classList.remove('active');
});

this.classList.add('active');
rankDipilih = this.innerText;

});
});

document.querySelectorAll('#mapOptions .option').forEach(opt=>{
opt.addEventListener('click',function(){

document.querySelectorAll('#mapOptions .option').forEach(o=>{
o.classList.remove('active');
});

this.classList.add('active');
mapDipilih = this.innerText;

});
});

document.getElementById('kategoriBayar').addEventListener('change',function(){

let metode = document.getElementById('metodeBayar');

metode.innerHTML = '<option value="">Pilih Metode Pembayaran</option>';

if(this.value == "E-Wallet"){

["Dana","GoPay","ShopeePay"].forEach(item=>{
metode.innerHTML += `<option value="${item}">${item}</option>`;
});

}else if(this.value == "Bank"){

["BSI","BRI","BNI","Bank Permata"].forEach(item=>{
metode.innerHTML += `<option value="${item}">${item}</option>`;
});

}

});

function randomOrder(){
return 'BOA-' + Math.random().toString(36).substring(2,10).toUpperCase();
}

function buatNota(){

const nama = document.getElementById('nama').value;
const idff = document.getElementById('idff').value;
const metode = document.getElementById('metodeBayar').value;
const rekening = document.getElementById('rekening').value;

if(!nama || !idff || !paketDipilih || !rankDipilih || !mapDipilih || !metode || !rekening){
alert("Harap lengkapi semua data!");
return;
}

const subtotal = hargaPaket + totalTambahan;
const orderId = randomOrder();

let fiturHTML = "";
let fiturWA = "";

fiturTambahan.forEach(f=>{
fiturHTML += `<li>${f.nama} (+Rp ${f.harga/1000}K)</li>`;
fiturWA += `- ${f.nama} (+Rp ${f.harga/1000}K)%0A`;
});

document.getElementById('nota').innerHTML = `

<div class="note">

<h2 style="text-align:center;color:#d8b4fe;">⚡ Battle of Legends</h2>

<p style="text-align:center;">
Selamat menyewa CR di Battle of Legends! 🎉
</p>

<br>

<p><b>ID Order:</b> ${orderId}</p>
<p><b>Nama Pemain:</b> ${nama}</p>
<p><b>ID Free Fire:</b> ${idff}</p>
<p><b>Rank:</b> ${rankDipilih}</p>
<p><b>Map Bermain:</b> ${mapDipilih}</p>
<p><b>Paket:</b> ${paketDipilih}</p>
<p><b>Detail Paket:</b> ${detailPaket}</p>

<p><b>Fitur Tambahan:</b></p>

<ul>
${fiturHTML || '<li>Tidak ada</li>'}
</ul>

<p><b>Metode Pembayaran:</b> ${metode}</p>

<button class="btn" onclick="kirimWA('${orderId}',${subtotal})">
📲 Kirim Pesanan via WhatsApp
</button>

</div>

}

const nama = document.getElementById('nama').value;
const idff = document.getElementById('idff').value;
const metode = document.getElementById('metodeBayar').value;
const rekening = document.getElementById('rekening').value;

const admins = [
'6285166024686',

];
